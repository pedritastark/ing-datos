CREATE TABLE Caso (
    codigo_caso numeric(10),
    codigo_crimen numeric(10),
    locacion character varying(50),
    codigo_arma numeric(10),
    codigo_delito numeric(10),
    codigo_premisa numeric(10),
    estado_crimen character varying(50),
    codigo_area numeric(10),
    PRIMARY KEY (codigo_caso)
);

CREATE TABLE Estado (
    estado_crimen character varying(50),
    descripcion_estado character varying,
    PRIMARY KEY (estado_crimen)
);

create table Area
(
	codigo_area numeric(10),
	nombre character varying,
	primary key(codigo_area)
);


create table Crimen
(
	codigo_crimen numeric(10),
	descripcion character varying not null,
	primary key(codigo_crimen)
);

create table Arma
(
	codigo_arma numeric(10),
	arma_usada character varying not null,
	primary key(codigo_arma)
);

create table Premisa
(
	codigo_premisa numeric(10),
	descripcion character varying not null,
	primary key(codigo_premisa)
);

create table Ubicacion
(
	locacion character varying (50),
	direccion character varying,
	latitud numeric,
	longitud numeric,	
	codigo_distrito numeric(10) not null,
	primary key(locacion)
);


create table Delito
(
	codigo_delito numeric(10),
	codigo_descripcion character varying not null,
	primary key(codigo_delito)
);

create table Delito_asociado
(
	codigo_delito_asociado numeric (50),
	codigo_distrito numeric(10) not null,
	direccion character varying not null,
	latitud numeric not null,
	longitud numeric not null,	
	primary key(codigo_delito_asociado)
);

create table Victima

(
	codigo_caso numeric(10),
	sexo_victima character varying not null,
	edad_victima numeric(10) not null,
	descendencia_victima character varying not null
);
